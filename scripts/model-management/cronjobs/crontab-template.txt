# Model Management System Cronjobs
# This file contains the scheduled tasks for the model management system
# Copy this file to /etc/cron.d/model-management or use it as a template for your crontab

# Environment variables
SHELL=/bin/bash
PATH=/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/sbin:/bin
NODE_ENV=production
LOG_DIR=/var/log/model-management

# Daily model synchronization at 2 AM
0 2 * * * root cd /path/to/aigens-backend && node scripts/model-management/cronjobs/sync-all-models.js --mode=update >> $LOG_DIR/daily-sync.log 2>&1

# Weekly full synchronization on Sundays at 1 AM
0 1 * * 0 root cd /path/to/aigens-backend && node scripts/model-management/cronjobs/sync-all-models.js --mode=full-sync >> $LOG_DIR/weekly-sync.log 2>&1

# Health check every 15 minutes
*/15 * * * * root cd /path/to/aigens-backend && node scripts/model-management/cronjobs/health-check.js >> $LOG_DIR/health-check.log 2>&1

# Log rotation daily at 3 AM
0 3 * * * root cd /path/to/aigens-backend && node scripts/model-management/cronjobs/rotate-logs.js >> $LOG_DIR/log-rotation.log 2>&1

# Metrics collection every 30 minutes
*/30 * * * * root cd /path/to/aigens-backend && node scripts/model-management/cronjobs/collect-metrics.js >> $LOG_DIR/metrics.log 2>&1

# Database backup daily at 4 AM
0 4 * * * root cd /path/to/aigens-backend && node scripts/model-management/cronjobs/backup-database.js >> $LOG_DIR/backup.log 2>&1
